## Command Structure

```bash
# Create a basic port scanner
python create_python_cli.py --type scanner --name portscan --target-range "1-1024"

# Create an exploit tool
python create_python_cli.py --type exploit --name web-exploit --protocol http --vulnerability "SQL Injection"

# Create a reconnaissance tool
python create_python_cli.py --type recon --name subdomain-enum --technique dns-bruteforce

# Create a custom security tool
python create_python_cli.py --type custom --name my-security-tool --description "Custom security analysis tool"
```

## Required Arguments

- `--name`: Tool name (used for class name and filename)
- `--type`: Tool category (scanner/exploit/recon/custom)
- `--description`: Brief description of the tool's purpose

## Optional Arguments

- `--target-type`: Type of target (ip/hostname/domain/url)
- `--protocol`: Network protocol (tcp/udp/http/https)
- `--technique`: Scanning/exploit technique
- `--output-format`: Output format (json/xml/text)
- `--threads`: Number of concurrent threads
- `--timeout`: Operation timeout in seconds
- `--verbose`: Enable verbose output
- `--safe-mode`: Enable safety checks and rate limiting

```python
#!/usr/bin/env python3
"""
Advanced Port Scanner
Author: Generated by ICWR Python CLI Tool
Description: Comprehensive port scanning tool with service detection
"""

import argparse
import socket
import sys
import threading
import time
from concurrent.futures import ThreadPoolExecutor, as_completed
import json
import xml.etree.ElementTree as ET

class AdvancedPortScanner:
    def __init__(self):
        self.parser = self.setup_arguments()
        self.args = self.parser.parse_args()

        # Common port services mapping
        self.services = {
            21: 'FTP', 22: 'SSH', 23: 'Telnet', 25: 'SMTP',
            53: 'DNS', 80: 'HTTP', 110: 'POP3', 143: 'IMAP',
            443: 'HTTPS', 993: 'IMAPS', 995: 'POP3S'
        }

        self.open_ports = []
        self.scan_results = {}

        self.run()

    def setup_arguments(self):
        parser = argparse.ArgumentParser(
            description="Advanced Port Scanner with Service Detection",
            formatter_class=argparse.RawDescriptionHelpFormatter,
            epilog="""
Examples:
  %(prog)s -t 192.168.1.1 -p 1-1024
  %(prog)s -t example.com -p 80,443,8080 --threads 50
  %(prog)s -t 10.0.0.0/24 --network-scan --output json
            """
        )

        parser.add_argument(
            '-t', '--target',
            required=True,
            help='Target IP address, hostname, or CIDR range'
        )

        parser.add_argument(
            '-p', '--ports',
            default='1-1024',
            help='Port range to scan (default: 1-1024)'
        )

        parser.add_argument(
            '--threads',
            type=int,
            default=100,
            help='Number of concurrent threads (default: 100)'
        )

        parser.add_argument(
            '--timeout',
            type=float,
            default=1.0,
            help='Connection timeout in seconds (default: 1.0)'
        )

        parser.add_argument(
            '--service-detection',
            action='store_true',
            help='Enable service version detection'
        )

        parser.add_argument(
            '--network-scan',
            action='store_true',
            help='Enable network range scanning'
        )

        parser.add_argument(
            '--output',
            choices=['text', 'json', 'xml'],
            default='text',
            help='Output format (default: text)'
        )

        parser.add_argument(
            '--output-file',
            help='Save results to file'
        )

        parser.add_argument(
            '--verbose',
            action='store_true',
            help='Enable verbose output'
        )

        parser.add_argument(
            '--safe-mode',
            action='store_true',
            help='Enable safe mode with rate limiting'
        )

        return parser

    def validate_target(self, target):
        """Validate target format"""
        import ipaddress

        try:
            # Check if it's a valid IP address
            ipaddress.ip_address(target)
            return True, target
        except ValueError:
            try:
                # Check if it's a valid hostname
                resolved_ip = socket.gethostbyname(target)
                return True, resolved_ip
            except socket.gaierror:
                return False, None

    def parse_ports(self, port_range):
        """Parse port range string"""
        ports = []
        for part in port_range.split(','):
            part = part.strip()
            if '-' in part:
                try:
                    start, end = map(int, part.split('-'))
                    ports.extend(range(start, end + 1))
                except ValueError:
                    continue
            else:
                try:
                    ports.append(int(part))
                except ValueError:
                    continue
        return sorted(list(set(ports)))

    def scan_port(self, port):
        """Scan a single port"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.args.timeout)
            result = sock.connect_ex((self.target_ip, port))

            if result == 0:
                service = self.detect_service(sock, port) if self.args.service_detection else self.services.get(port, 'Unknown')
                self.open_ports.append(port)
                self.scan_results[port] = {
                    'state': 'open',
                    'service': service,
                    'timestamp': time.time()
                }

                if self.args.verbose:
                    print(f"[+] Port {port}/tcp open - {service}")

            sock.close()

        except Exception as e:
            if self.args.verbose:
                print(f"[-] Error scanning port {port}: {str(e)}")

    def detect_service(self, sock, port):
        """Basic service detection"""
        try:
            if port == 80:
                sock.send(b"HEAD / HTTP/1.0\r\n\r\n")
                response = sock.recv(1024).decode('utf-8', errors='ignore')
                if 'HTTP' in response:
                    return 'HTTP'
            elif port == 443:
                return 'HTTPS'
            elif port in [21, 22, 25, 110, 143]:
                banner = sock.recv(1024).decode('utf-8', errors='ignore').strip()
                if banner:
                    return f"{self.services.get(port, 'Unknown')} ({banner[:50]})"
        except:
            pass

        return self.services.get(port, 'Unknown')

    def rate_limit_delay(self):
        """Implement rate limiting"""
        if self.args.safe_mode:
            time.sleep(0.01)  # 10ms delay between scans

    def run_scan(self):
        """Execute the port scan"""
        print(f"[+] Starting scan against {self.args.target}")
        print(f"[+] Ports: {self.args.ports}")
        print(f"[+] Threads: {self.args.threads}")
        print(f"[+] Timeout: {self.args.timeout}s")
        if self.args.safe_mode:
            print("[+] Safe mode enabled")
        print("-" * 60)

        ports = self.parse_ports(self.args.ports)
        if not ports:
            print("[-] No valid ports specified")
            return

        start_time = time.time()

        with ThreadPoolExecutor(max_workers=self.args.threads) as executor:
            futures = [executor.submit(self.scan_port, port) for port in ports]

            for future in as_completed(futures):
                self.rate_limit_delay()

        end_time = time.time()
        scan_duration = end_time - start_time

        self.display_results(scan_duration)

    def display_results(self, duration):
        """Display scan results"""
        print("-" * 60)
        print(f"[+] Scan completed in {duration:.2f} seconds")
        print(f"[+] Found {len(self.open_ports)} open ports")

        if self.open_ports:
            print("\nOpen ports:")
            for port in sorted(self.open_ports):
                service = self.scan_results.get(port, {}).get('service', 'Unknown')
                print(f"  {port}/tcp - {service}")

        # Save results if requested
        if self.args.output_file:
            self.save_results()

    def save_results(self):
        """Save results to file"""
        if self.args.output == 'json':
            with open(self.args.output_file, 'w') as f:
                json.dump({
                    'target': self.args.target,
                    'open_ports': self.open_ports,
                    'results': self.scan_results,
                    'scan_time': time.time()
                }, f, indent=2)

        elif self.args.output == 'xml':
            root = ET.Element("scan_results")
            target_elem = ET.SubElement(root, "target")
            target_elem.text = self.args.target

            ports_elem = ET.SubElement(root, "open_ports")
            for port in self.open_ports:
                port_elem = ET.SubElement(ports_elem, "port")
                port_elem.set("number", str(port))
                port_elem.set("service", self.scan_results.get(port, {}).get('service', 'Unknown'))

            tree = ET.ElementTree(root)
            tree.write(self.args.output_file)

    def run(self):
        """Main execution method"""
        print("=" * 80)
        print("           Advanced Port Scanner")
        print("           Generated by ICWR Python CLI Tool")
        print("=" * 80)

        # Validate target
        valid, resolved_ip = self.validate_target(self.args.target)
        if not valid:
            print(f"[-] Invalid target: {self.args.target}")
            sys.exit(1)

        self.target_ip = resolved_ip

        try:
            self.run_scan()
        except KeyboardInterrupt:
            print("\n[!] Scan interrupted by user")
            sys.exit(0)
        except Exception as e:
            print(f"[-] Error during scan: {str(e)}")
            if self.args.verbose:
                import traceback
                traceback.print_exc()
            sys.exit(1)

if __name__ == "__main__":
    AdvancedPortScanner()
```
